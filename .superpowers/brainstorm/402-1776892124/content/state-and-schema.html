Job-Zustandsautomat und SQLite-Schema

Die Quelle der Wahrheit — beide Prozesse (Webserver, Worker) lesen und schreiben ausschließlich über dieses Schema.

       │     └──────────────────────┘                          │
       │                   │                                   │
       │            [POST /export]                             │
       │                   ▼                                   │
       │            ┌─────────────┐                            │
       │            │  EXPORTED   │                            ▼
       │            └─────────────┘                     ┌──────────┐
       │                                                │ FAILED   │
       └────────────────────────────────────────────────┤ + error  │
                                                        └──────────┘

Nur der Worker darf Status verändern (außer Webserver: PENDING beim Erstellen, EXPORTED beim Export-Trigger). Bei jedem Worker-Start: alle Jobs in DOWNLOADING/TILING/INFERRING werden auf FAILED mit Grund "worker_restarted" gesetzt.

SQLite-Schema

-- Initialisierung jeder Connection
PRAGMA journal_mode = WAL;
PRAGMA busy_timeout = 5000;
PRAGMA synchronous = NORMAL;
PRAGMA foreign_keys = ON;

CREATE TABLE jobs (
    id                TEXT PRIMARY KEY,              -- UUID v4
    prompt            TEXT NOT NULL,
    bbox_wgs84        TEXT NOT NULL,                 -- JSON: [minx,miny,maxx,maxy]
    bbox_utm_snapped  TEXT NOT NULL,                 -- JSON, grid-aligned 0.2 m
    status            TEXT NOT NULL,                 -- siehe State Machine
    error_reason      TEXT,                          -- enum: OOM, WCS_FAIL, ...
    error_message     TEXT,                          -- freier Text + Stacktrace-Tail
    dop_vrt_path      TEXT,                          -- nach Download gesetzt
    gpkg_path         TEXT,                          -- nach Export gesetzt
    tile_total        INTEGER,
    tile_completed    INTEGER DEFAULT 0,
    tile_failed       INTEGER DEFAULT 0,
    created_at        TEXT NOT NULL,                 -- ISO-8601
    started_at        TEXT,
    finished_at       TEXT
);
CREATE INDEX idx_jobs_status_created ON jobs(status, created_at);

CREATE TABLE polygons (
    id                INTEGER PRIMARY KEY AUTOINCREMENT,
    job_id            TEXT NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
    geometry_wkt      TEXT NOT NULL,                 -- WKT in EPSG:25832
    score             REAL NOT NULL,                 -- SAM 3.1 Detection Score
    source_tile_row   INTEGER NOT NULL,              -- für Debugging/Traceability
    source_tile_col   INTEGER NOT NULL,
    validation        TEXT                            -- NULL | ACCEPTED | REJECTED
);
CREATE INDEX idx_polygons_job ON polygons(job_id);

CREATE TABLE nodata_regions (
    id                INTEGER PRIMARY KEY AUTOINCREMENT,
    job_id            TEXT NOT NULL REFERENCES jobs(id) ON DELETE CASCADE,
    geometry_wkt      TEXT NOT NULL,                 -- Kachel-Footprint in EPSG:25832
    tile_row          INTEGER NOT NULL,
    tile_col          INTEGER NOT NULL,
    reason            TEXT NOT NULL                  -- OOM | WCS_TIMEOUT | SEG_FAULT | ...
);
CREATE INDEX idx_nodata_job ON nodata_regions(job_id);
